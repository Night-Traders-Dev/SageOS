def memoize(func):
    cache = {}
    def wrapper(*args):
        if args not in cache:
            cache[args] = func(*args)
        return cache[args]
    return wrapper

@memoize
def fib(n):
    if n < 2:
        return n
    return fib(n-1) + fib(n-2)

// Linked list
typedef struct Node {
    int data;
    struct Node* next;
} Node;

Node* list_push(Node* head, int data) {
    Node* node = (Node*)malloc(sizeof(Node));
    node->data = data;
    node->next = head;
    return node;
}

void list_free(Node* head) {
    while (head) {
        Node* next = head->next;
        free(head);
        head = next;
    }
}

int list_length(Node* head) {
    int count = 0;
    while (head) { count++; head = head->next; }
    return count;
}

// Hash table
#define TABLE_SIZE 256

typedef struct Entry {
    char* key;
    void* value;
    struct Entry* next;
} Entry;

typedef struct {
    Entry* buckets[TABLE_SIZE];
    int count;
} HashTable;

unsigned int hash(const char* key) {
    unsigned int h = 5381;
    while (*key) h = ((h << 5) + h) + (unsigned char)*key++;
    return h % TABLE_SIZE;
}

void ht_set(HashTable* ht, const char* key, void* value) {
    unsigned int idx = hash(key);
    Entry* e = (Entry*)malloc(sizeof(Entry));
    e->key = strdup(key);
    e->value = value;
    e->next = ht->buckets[idx];
    ht->buckets[idx] = e;
    ht->count++;
}

void* ht_get(HashTable* ht, const char* key) {
    unsigned int idx = hash(key);
    Entry* e = ht->buckets[idx];
    while (e) {
        if (strcmp(e->key, key) == 0) return e->value;
        e = e->next;
    }
    return NULL;
}

// Memory arena allocator
typedef struct {
    char* base;
    size_t size;
    size_t offset;
} Arena;

Arena arena_create(size_t size) {
    Arena a;
    a.base = (char*)malloc(size);
    a.size = size;
    a.offset = 0;
    return a;
}

void* arena_alloc(Arena* a, size_t size) {
    size_t aligned = (a->offset + 7) & ~7;
    if (aligned + size > a->size) return NULL;
    void* ptr = a->base + aligned;
    a->offset = aligned + size;
    return ptr;
}

void arena_reset(Arena* a) { a->offset = 0; }
void arena_destroy(Arena* a) { free(a->base); }

// Bit manipulation
int count_bits(unsigned int n) {
    int count = 0;
    while (n) { count += n & 1; n >>= 1; }
    return count;
}

int is_power_of_two(unsigned int n) {
    return n && !(n & (n - 1));
}

unsigned int next_power_of_two(unsigned int n) {
    n--;
    n |= n >> 1; n |= n >> 2; n |= n >> 4;
    n |= n >> 8; n |= n >> 16;
    return n + 1;
}

// Smart pointers
class TreeNode {
public:
    int value;
    std::unique_ptr<TreeNode> left;
    std::unique_ptr<TreeNode> right;

    TreeNode(int v) : value(v) {}

    void insert(int v) {
        if (v < value) {
            if (left) left->insert(v);
            else left = std::make_unique<TreeNode>(v);
        } else {
            if (right) right->insert(v);
            else right = std::make_unique<TreeNode>(v);
        }
    }
};

# Cross-Language Comparison: The Same Algorithm in 5 Languages

## Quicksort

### Python
def quicksort(arr):
    if len(arr) <= 1:
        return arr
    pivot = arr[len(arr) // 2]
    left = [x for x in arr if x < pivot]
    middle = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]
    return quicksort(left) + middle + quicksort(right)

### C
void quicksort(int* arr, int low, int high) {
    if (low >= high) return;
    int pivot = arr[(low + high) / 2];
    int i = low, j = high;
    while (i <= j) {
        while (arr[i] < pivot) i++;
        while (arr[j] > pivot) j--;
        if (i <= j) {
            int tmp = arr[i]; arr[i] = arr[j]; arr[j] = tmp;
            i++; j--;
        }
    }
    quicksort(arr, low, j);
    quicksort(arr, i, high);
}

### C++
template<typename T>
void quicksort(std::vector<T>& arr, int low, int high) {
    if (low >= high) return;
    T pivot = arr[(low + high) / 2];
    int i = low, j = high;
    while (i <= j) {
        while (arr[i] < pivot) i++;
        while (arr[j] > pivot) j--;
        if (i <= j) { std::swap(arr[i++], arr[j--]); }
    }
    quicksort(arr, low, j);
    quicksort(arr, i, high);
}

### Nim
proc quicksort[T](arr: var seq[T], low, high: int) =
  if low >= high: return
  let pivot = arr[(low + high) div 2]
  var i = low
  var j = high
  while i <= j:
    while arr[i] < pivot: inc i
    while arr[j] > pivot: dec j
    if i <= j:
      swap(arr[i], arr[j])
      inc i
      dec j
  quicksort(arr, low, j)
  quicksort(arr, i, high)

### Sage
proc quicksort(arr, low, high):
    if low >= high:
        return
    let pivot = arr[((low + high) / 2) | 0]
    let i = low
    let j = high
    while i <= j:
        while arr[i] < pivot:
            i = i + 1
        while arr[j] > pivot:
            j = j - 1
        if i <= j:
            let tmp = arr[i]
            arr[i] = arr[j]
            arr[j] = tmp
            i = i + 1
            j = j - 1
    quicksort(arr, low, j)
    quicksort(arr, i, high)
